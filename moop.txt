bool cmp(pair<ll, ll> x, pair<ll, ll> y)
{
	if (x.first == y.first)
	{
		//x should go before y if x.second > y.second
		return (x.second > y.second);
	}
	return (x.first < y.first);
}


int main()
{
	setIO("moop");
	cin >> n;
	fi(n)
	{
		cin >> a >> b;
		points.pb({ a,b });
	}
	sort(all(points));
	vector<ll> temp;
	vector<ll>::iterator itr;
	temp.push_back(points[0].second);
	for (int i = 1; i < n; i++)
	{
		itr = upper_bound(all(temp), points[i].second);
		if (itr == temp.end())
		{
			temp.erase(temp.begin() + 1, temp.end());
		}
		else if (itr == temp.begin())
		{
			temp.insert(temp.begin(), points[i].second);
		}
		else
		{
			temp.erase(temp.begin() + 1, itr);
		}
	}
	cout << temp.size() << endl;
}
